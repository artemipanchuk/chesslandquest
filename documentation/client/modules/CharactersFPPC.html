@define 'Characters/FPPC', (exports) ->
  [CanvasCharacters, CanvasControls] =
    @require 'Canvas/Characters',
             'Canvas/Controls'

  LocationManager =
    @require 'Location/Manager'

  Socket =
    @require 'Socket'

  self = undefined

  exports.register = (character) ->
    self = character

    CanvasControls.addMouseAction 'click', (to) ->
      from = self.point
      path = LocationManager.createPath from, to

      return unless path

      Socket.send 'fppc:move', to

      CanvasCharacters.move 'fppc', self, path

  exports.spawn = (point) ->
    self.point = point

    Socket.send 'fppc:register', point

    CanvasCharacters.register 'fppc', self

  exports.get = ->
    return self
