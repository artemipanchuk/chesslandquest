@define 'Interface/HUD/Dialogs', (exports) ->
  dialogs =
    'npc' : ({name}) =>
      CharactersNPC =
        @require 'Characters/NPC'

      opponent = CharactersNPC.get name

      dialog = dom
        .create('section')
        .attr('id', 'dialog')
        .appendTo(document.body)

      dom
        .create('button')
        .attr('id', 'buttonClose')
        .appendTo(dialog)
        .bind 'click', ->
          dialog.destroy()

      dom
        .create('h2')
        .text("#{name.ucfirst()} the Computer")
        .appendTo(dialog)

      sectionUnits = dom
        .create('section')
        .attr('id', 'sectionUnits')
        .appendTo(dialog)

      dom
        .create('button')
        .attr('id', 'buttonFight')
        .text('Fight!')
        .appendTo(dialog)
        .bind 'click', =>
          dialog
            .addClass('waiting')
            .find('*')
            .destroy()

          dom
            .create('h2')
            .text('Please, wait')
            .appendTo(dialog)

          Socket.send 'battle:call', {'npc', name}

      units = {}
      for unit in opponent.army
        units[uname = unit.stats.name] ?= 0
        ++units[uname]

      for unit, quantity of units
        dom
          .create('span')
          .addClass('sectionUnitsName')
          .text(unit)
          .appendTo(sectionUnits)

        dom
          .create('span')
          .addClass('sectionUnitsQuantity')
          .text(quantity)
          .appendTo(sectionUnits)

    'tppc' : ({name}) =>
      Application =
        @require 'Application'

      CharactersTPPC =
        @require 'Characters/TPPC'

      opponent = CharactersTPPC.get name

      console.dir opponent

      dialog = dom
        .create('section')
        .attr('id', 'dialog')
        .appendTo(document.body)

      dom
        .create('button')
        .attr('id', 'buttonClose')
        .appendTo(dialog)
        .bind 'click', ->
          dialog.destroy()

      dom
        .create('h2')
        .text("General #{name.ucfirst()}")
        .appendTo(dialog)

      sectionUnits = dom
        .create('section')
        .attr('id', 'sectionUnits')
        .appendTo(dialog)

      dom
        .create('button')
        .attr('id', 'buttonFight')
        .text('Fight!')
        .appendTo(dialog)
        .bind 'click', =>
          dialog
            .find('*')
            .destroy()

          dom
            .create('h2')
            .text('Please, wait')
            .appendTo(dialog)

          dialog.addClass 'waiting'

          Application.initiateBattle 'tppc', name

      units = {}

      units.general = "#{name.ucfirst()} of #{opponent.experience.rank}"

      for unit in opponent.army
        units[unitName = unit.stats.name] ?= 0
        ++units[unitName]

      for unit, quantity of units
        dom
          .create('span')
          .addClass('sectionUnitsName')
          .text(unit)
          .appendTo(sectionUnits)

        dom
          .create('span')
          .addClass('sectionUnitsQuantity')
          .text(quantity)
          .appendTo(sectionUnits)

    'battle:reject' : ->
      dom('section#dialog')
        .removeClass('waiting')
        .find('*')
        .destroy()
        .append dom
          .create('h2')
          .text('Battle has been refused')
          .addClass('centered')

    dialogs[type]? name
