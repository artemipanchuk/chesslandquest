  exports.initialize = (app) ->
    setupCharactersLayer app

  exports.register = (type, {name, point}) ->
    elements["#{type}:#{name}"] = new Element layer,
      {name, point}, type

  exports.drop = (type, name) ->
    elements["#{type}:#{name}"]

  exports.move = (type, character, path) ->
    {name} = character

    elements["#{type}:#{name}"]?.move path, (point) ->
      character.point = point
