@define 'Application', (exports) ->
  [CharactersManager, InterfaceManager, LocationManager, CanvasManager] =
    @require 'Characters/Manager',
             'Interface/Manager',
             'Location/Manager',
             'Canvas/Manager'

  Socket =
    @require 'Socket'

  initiateGame = (state) ->
    LocationManager  .initialize state.character.location, state.location
    CharactersManager.initialize state.character
    CharactersManager.spawn LocationManager.getSpawn()

  listen = ->
    Socket.listen 'state', initiateGame

    Socket.listen 'battle:answer', (battle) ->
      if battle.status is true
        InterfaceManager.buildBattleInterface ->
      else
        BattleManager.reject battle

    Socket.listen 'battle:call', (battle) ->
      BattleManager.notify battle
