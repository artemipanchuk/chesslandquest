    clearPrevious : (ctx) ->

    renderTo : (ctx, resources) ->

    move : (path, onstep) ->
      {name, shape} = @

      Animations.reset name

      redraw = @redraw.bind @

      for step in path then do ->
        vector = undefined
        target = Projection.translatePoint step

        next = step

        Animations.add
          target : name
          length : 250

          pre : (part) ->
            vector = shape.center.diff(target).mul(part)

          tick : ->
            shape.move vector

            do redraw

          finish : ->
            onstep next
