@define 'Characters/NPC', (exports) ->
  [CanvasCharacters, CanvasControls] =
    @require 'Canvas/Characters',
             'Canvas/Controls'

  LocationManager =
    @require 'Location/Manager'

  Socket =
    @require 'Socket'

  all = {}

  listen = ->
    Socket.listen 'npc:register', (character) ->
      register character

    Socket.listen 'npc:drop', (name) ->
      drop name

    Socket.listen 'npc:move', (course) ->
      move course

  register = (character) ->
    all[character.name] = character

    CanvasCharacters.register 'npc', character

  move = (course) ->
    {name, point} = course

    character = all[name]

    from = character.point
    to   = point

    path = LocationManager.createPath from, to

    return unless path

    CanvasCharacters.move 'npc', character, path

  drop = (name) ->
    CanvasCharacters.drop 'npc', name

    delete all[name]
