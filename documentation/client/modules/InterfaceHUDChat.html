Private

 
  messages = undefined
  field    = undefined

  unsafe = /[^a-zA-Zа-яА-Я.,!?—\- ]+/g
  escape = (string) ->
    if unsafe.test string
      return string.replace unsafe, ''
    else
      return string

  post = (sender, text) ->
    dom
      .create('section')
      .addClass('message')
      .text("#{sender}: #{text}")
      .appendTo messages

  send = ->
    text = escape field.text()

    return unless text.trim().length

    field.text ''

    post 'Me', text

    Socket.send 'chat', text

  listen = ->
    Socket.listen 'chat', (message) ->
      post message.name, message.text

  exports.build = ->
    chat = dom
      .create('section')
      .attr('id', 'chat')
      .appendTo document.body

    dom
      .create('section')
      .addClass('header')
      .appendTo chat

    messages = dom
      .create('section')
      .addClass('messages')
      .appendTo chat

    field = dom
      .create('section')
      .addClass('field')
      .attr('contenteditable', 'true')
      .appendTo chat

    chat.bind 'keypress', (event) ->
      if event.keyCode is 13
        do send
