Drawing
             		                            
  projectTile = (settings) ->
    {texture, cell, ctx} = settings

    polygon = Projection.translateTile cell.rectangle.toPolygon()

    tilesLayer.ctx.projectiveImage
      image : Preloader.get texture
      to    : polygon

  engine.setMethod
    grass : (ctx, cell) ->
      projectTile
        texture : "grass#{variations 1}"
        cell    : cell

    earth : (ctx, cell) ->
      projectTile
        texture : "earth#{variations 1}"
        cell    : cell

    road : (ctx, cell) ->
      projectTile
        texture : "road#{variations 1}"
        cell    : cell

  exports.projectTile = (settings) ->
    {tile, x, y} = settings

    engine.getCellByIndex(new Point x, y).value = tile

  exports.projectTransition = (settings) ->
    {x, y, angle, texture} = settings

    polygon = engine.getCellByIndex(new Point x, y).rectangle.toPolygon()

    polygon.rotate angle, polygon.center

    polygon = Projection.translatePolygon polygon

    propsLayer.ctx.projectiveImage
      image : Preloader.get texture
      to    : polygon

  exports.projectObject = (settings) ->
    {name:texture, x, y} = settings

    polygon = engine.getCellByIndex(new Point x, y).rectangle.toPolygon()

    image = Preloader.get texture

    from = Projection.translatePoint polygon.center
    from.move new Point -image.width+25, -image.height+25

    object = new Rectangle
      from : from
      size :
        width  : image.width
        height : image.height

    polygon = object.toPolygon()

    objectsLayer.ctx.projectiveImage
      image : image
      to    : polygon

  exports.getCellOf = (point) ->
    coordinates = point.clone().mul(size)

  exports.getCellByPoint = (point) ->
    return engine.getCellByPoint new Point point

  exports.redraw = ->
    for cell in engine.cells
      engine.updateCell cell
