  exports.translatePoint = (point3D) ->
    if point3D instanceof Array
      point3D = new Point3D point3D.first, point3D.second

    point2D = projection.toIsometric point3D

    return point2D

  exports.translateTile = (polygon) ->
    polygon.scale 1.1, polygon.center

    polygon.points = polygon.points.map (point2D) ->
      return projection.toIsometric point2D

    return polygon
