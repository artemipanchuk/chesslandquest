About tiles
             		                            
             		           
                               
                            
                 ¶               
               
Every tile could have many variations
             		                            
             		           
                               
                            
                 ¶               
               
Note, that there are 2 types of tiles:
             		                            
             		           
                               
                            
                 ¶               
               
Weak
             		                            
weak = ['grass', 'cobble']
             		           
                               
                            
                 ¶               
               
Strong
             		                            
strong = ['earth', 'road']
             		           
                               
                            
                 ¶               
               
All strong tiles have their 2 (side and corner) special transition tiles,
which applies to neighbor weak tiles
             		                            
             		           
                               
                            
                 ¶               
               
Weak tiles are usual tiles

• Process location data;



code (string)
             		                            
  process = (location) ->
    {matrix} = location

    height = matrix.length
    width  = matrix.first.length

    Pathfinder.update matrix, width, height

    for row, y in matrix
      for value, x in row
        tile = switch value
          when 'g' then 'grass'
          when 'e' then 'earth'
          when 'r' then 'road'

        if (strong.some (strong) -> tile is strong)
          processStrong tile, x, y, matrix
        else
          processWeak   tile, x, y

    CanvasLocation.redraw()

    for object in location.objects
      processObject object

    markers = {}

    for type, marker of location.markers
      markers[type] = marker

• Process strong tile;



tile (string), x (number, y (number), matrix (array[][])
             		                            
  processStrong = (tile, x, y, matrix) ->
    CanvasLocation.projectTile {tile, x, y}

    sides   = getPotentialSides   tile,  x, y, matrix
    corners = getPotentialCorners tile,  x, y, sides, matrix

    for transitions, index in [sides, corners]
      for position, cell of transitions
        angle = switch position
          when '1' then  π
          when '2' then -π/2
          when '3' then  0
          when '4' then  π/2

        type = switch index
          when 0 then 's'
          when 1 then 'c'

        CanvasLocation.projectTransition
          x : cell.x
          y : cell.y

          angle   : angle
          texture : "#{tile}#{type}"

  processWeak = (tile, x, y) ->

  processObject = (settings) ->
    {width, height} = settings

    if width and height
      for x in [0..width]
        for y in [0..height]
          closed.push {x, y}
    else
      {x, y} = settings

      Pathfinder.close x, y

    CanvasLocation.projectObject settings

  {size} = CanvasLocation
  exports.createPath = (from, to) ->
    exactX = to.x%size
    exactY = to.y%size

    from = CanvasLocation.getCellByPoint(new Point from)?.point
    to   = CanvasLocation.getCellByPoint(new Point to  )?.point

    return unless from and to

    path = Pathfinder.find from, to

    return no unless path.length

    path = path.map (step) ->
      return CanvasLocation.getCellOf step

    path.last.first += exactX
    path.last.last  += exactY

    path = path[1...]

    return path

name (string), location (object)
             		                            
  exports.initialize = (name, location) ->
    actual = name

    if cached = locations[name]
      process cached
    else
      locations[name] = location

      process location

  exports.getSpawn = ->
    return markers.spawn
