idPattern = /[^;=]+$/
ensureSession = (request) ->
  {cookie} = request.headers

  return no unless cookie

  cookies = {}

  for f, i in array = cookie.split /\=|;\s?/ by 2
    cookies[f] = array[i+1]

  session = cookies.id

  if session and AccountsManager.checkActuality session
    return session
  else
    return no

responseHTML = (path, response) ->
  response.statusCode = 200
  response.setHeader 'content-type', 'text/html'

  Ut.responseFile path, response

responseMessage = (message, response, code = 200) ->
  response.statusCode = code

  Ut.responseText message, response

responseJSON = (data, response, code = 200) ->
  message = JSON.stringify data

  responseMessage message, response, code

Routes =
  '/' : (request, response) ->
    session = ensureSession request

    if session
      redirect '/game', response
    else
      responseHTML 'client/pages/index.html', response

  '/signin/data' : (request, response) ->
    if ensurePost request
      request.addListener 'data', (chunk) ->
        form   = ensureIncomingData chunk

        unless form
          redirect '/406', response
          return

        status = FormParser.parseSignin form

        if status is true
          [username, password] =
            form[0..1].map (field) ->
              return field.value

          responseSession {username, password}, response
        else
          responseJSON status, response, 406
    else
      redirect '/403', response

  '/signup' : (request, response) ->
    responseHTML 'client/pages/signup.html', response

  '/signup/data' : (request, response) ->
    if ensurePost request
      request.addListener 'data', (chunk) ->
        form   = ensureIncomingData chunk

        unless form
          redirect '/406', response
          return

        status = FormParser.parseSignup form

        if status is true
          [username, password, date] =
            [form[0], form[1], form[3]].map (field) ->
              return field.value

          AccountsManager.createAccount {username, password, date}

          responseSession {username, password}, response
        else
          responseJSON status, response, 406
    else
      redirect '/403', response

  '/signout' : (request, response) ->
    session = ensureSession request

    if session
      AccountsManager.closeSession session

      response.statusCode = 200

      do response.end
    else
      redirect '/401', response

  '/game' : (request, response) ->
    session = ensureSession request

    if session
      responseHTML 'client/pages/game.html', response
    else
      redirect '/', response

  '/game/characters/my' : (request, response) ->
    session = ensureSession request

    unless session
      redirect '/401', response
      return

    username   = AccountsManager  .discoverUsernameById session
    characters = CharactersManager.getCharactersByOwner username

    responseJSON characters, response

  '/game/characters/create' : (request, response) ->
    if ensurePost request
      request.addListener 'data', (chunk) ->
        session = ensureSession request

        unless session
          redirect '/401', response
          return

        form = ensureIncomingData chunk

        unless form
          redirect '/403', response
          return

        status = FormParser.parseCharacterCreation form

        if status is true
          name  = form[0].value
          owner = AccountsManager.discoverUsernameById session

          CharactersManager.createCharacter {name, owner}

          responseCharacter name, response
        else
          responseJSON status, response, 406
    else
      redirect '/403', response

  '/game/characters/select' : (request, response) ->
    if ensurePost request
      request.addListener 'data', (chunk) ->
        session = ensureSession request

        unless session
          redirect '/401', response
          return

        form = ensureIncomingData chunk

        unless form
          redirect '/406', response
          return

        status = FormParser.parseCharacterSelection form

        if status is true
          name = form[0].value

          responseCharacter name, response
        else
          responseJSON status, response, 406
    else
      redirect '/403', response

  '/401' : (request, response) ->
    responseHTML 'client/pages/401.html', response

  '/403' : (request, response) ->
    responseHTML 'client/pages/403.html', response

  '/404' : (request, response) ->
    responseHTML 'client/pages/404.html', response

  'resource' : (path, request, response) ->
    extension   = (extensionPattern.exec path)?[0]
    contentType = contentTypeTable[extension] or 'text/plain'

    path = path.replace /\//, ''

    response.setHeader 'content-type', contentType

    Ut.responseResource path, request, response
