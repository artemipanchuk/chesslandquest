rand = (n) ->
  Math.round(Math.random()*n)

create =
  first : (npc, callback) ->
    {name, from, to, x, y} = npc

    exact = from + rand(to-from)
    point = {x, y}

    army = []

    exact.times (i) ->
      UnitsManager.getUnit 'private', (unit) ->
        {exports:unit} = unit

        army.push unit

        if army.length is exact
          callback {name, army, point}

NonPlayerCharactersManager =
  generate : (callback) ->
    locations = LocationsManager.getLocationsList()

    for locationName in locations then do ->
      LocationsManager.getLocation locationName, (location) ->
        {npcs} = location.internal

        for npc in npcs then do ->
          create[rank = npc.rank] npc, (created) ->
            created.location = locationName

            all.push created

            callback created
