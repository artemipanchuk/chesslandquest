Checking version key
             		                            
    Ut.version path, (actual) ->
      if (not location = cached[name]) or location.version isnt actual
        Fs.readFile path, 'utf8', (error, content) ->
          data = parse content

          location = cached[name] =
            internal : data.private
            exports  : data.public
            version  : actual

          callback location
      else
        callback location
