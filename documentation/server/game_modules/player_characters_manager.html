Precaching characters for further usage
             		                            
Db.within 'characters', (items) ->
  all = items.map (item) ->
    object = {}

    for property of item
      object[property] = item[property] if property isnt '_id'

    return object

  createCharacter : (character) ->
    log "Creating character: #{character.name} — #{character.owner}"

    for property, value of defaults

• Get character by it's name (passive or active);



name (string), active (boolean)
             		                            
  getCharacterByName : (name, active = no) ->
    target = undefined

    if active
      if id = hashTable[name]
        target = online[id]
    else
      all.some (character) ->
        if name is character.name
          target = character

          return true

    if target
      return target
    else
      log "DB: Corrupted character. Interrupting.", 'critical'

      return null

Internal mechanix
             		                            
  update : (name, property, value) ->
    return unless id = hashTable[name]

    (character = online[id])[property] = value

    {name} = character

    Db.update 'characters', {name}, character

• Select character for account to use;



name (string)
             		                            
  selectCharacter : (name) ->
    target = undefined

    all.some (character) ->
      if character.name is name
        target = character

        return true

    if target
      id = Ut.generateId()

      online[id] = target

      hashTable[name] = id

      log "Select — #{name} with #{id}"

      return id
    else
      log "DB: Corrupted character. Interrupting.", 'critical'

      return null
