      initiator = BattlesManager.getWaiter   pc.name

        battle = BattlesManager.createBattleWithNPC initiator, opponent

        wsi = findPC initiator.name

        ws.send "battle:answer #{JSON.stringify battle}" for ws in [wsi, wso]

getPCsNear = (initiator) ->
  vector = []

  for sec, pc of pcs
    if pc.id isnt initiator.id and pc.location is initiator.location
      vector.push PCManager.getCharacterById pc.id

  return vector

getNPCsNear = (initiator) ->
  vector = []

  for name, npc of npcs
    if npc.location is initiator.location
      vector.push npc

  return vector

findPC = (name) ->
  for sec, pc of pcs when pc.name is name then return pc

  return null

distribute = (subject, data, initiator = {}) ->
  pack = JSON.stringify data

  if initiator
    for sec, pc of pcs
      if pc.id isnt initiator.id
        pc.ws.send "#{subject} #{pack}"
  else
    for sec, pc of pcs then pc.ws.send "#{subject} #{pack}"

ensureSession = (ws) ->
  {cookie} = ws.upgradeReq.headers

  return no unless cookie

  cookies = {}

  for f, i in array = cookie.split /\=|;\s?/ by 2
    cookies[f] = array[i+1]

  session = cookies.id

  if session and AccountsManager.checkActuality session
    return yes
  else
    return no

Handling
             		                            
bySpace = ' '
byColon = ':'
handle = (message, ws) ->
  incoming = message.split bySpace

  unless pc = pcs[key ws]
    drop ws ; log error.data, 'critical'

    return

  command = incoming[0].split byColon

  if command.length is 1
    process = requests[command[0]]
  else
    process = requests[command[0]]?[command[1]]

  unless process
    drop ws ; log error.data, 'critical'

    return

  if incoming.length > 2
    chunk = incoming[1...].join bySpace
  else
    chunk = incoming[1]

  try
    data = JSON.parse chunk

  catch exception
    drop ws ; log error.json, 'critical'

    return

  process data, pc, ws

drop = (ws) ->
  ws.send 'Client has provocated an error' ; do ws.close

GAMEServer =
  use : (HTTPServer) ->
    log 'GAME: Running on 80', 'positive'

    NPCManager.generate (npc) ->
      npcs[npc.name] = npc

    (new WebSocketServer server : HTTPServer).on 'connection', (ws) ->
      unless ensureSession ws
        drop ws

        return

      pcs[key ws] = {}

      ws.on 'message', (message) ->
        handle message, ws

      ws.on 'close', ->
        id = pcs[sec = key ws].id

        delete pcs[sec]

        distribute 'tppc:drop', PCManager.getCharacterNameById id if id
