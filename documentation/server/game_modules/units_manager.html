read = (direction, n) ->
  return switch direction
    when 't' then {x: 0, y:-n}
    when 'r' then {x: n, y: 0}
    when 'b' then {x: 0, y: n}
    when 'l' then {x:-n, y: 0}

    when 'tr' then {x: n, y:-n}
    when 'br' then {x: n, y: n}
    when 'bl' then {x:-n, y: n}
    when 'tl' then {x:-n, y:-n}
