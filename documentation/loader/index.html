@handlers =
  'onfinish' : ->
    {dom} = atom

    Validator =
      @require 'Validator'

    form = dom '#loginForm'

    form.bind 'keypress', (event) ->
      if event.keyCode is 13
        Validator.submitSignin form

    form.find('button').bind 'click', ->
      Validator.submitSignin form
